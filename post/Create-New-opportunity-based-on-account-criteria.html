Trigger


trigger createOpportunity on Account (after insert) {
    List oppsToUpdate = new List();
    for(Account ac : Trigger.new) {
    if (ac.Rating== 'Hot') {
    Opportunity o = new Opportunity (); 
    o.StageName = 'Closed Won'; 
    o.CloseDate = Date.today(); 
    o.Name = 'Opp';
    oppsToUpdate.add(o);
    }
    
    }
    insert oppsToUpdate; 
    }
